// stack, use linked list to implement a stack
// a--> b--> c   suppose c is the top of the stack, if a is the top of the stack, it would be different.
class Stack{
	LinkedListNode head;
	
	
	Object pop(){
		if(head == null)
			return null;
		LinkedListNode runner = head;
		while(runner.next != null){
			runner = runner.next;
		}
		return runner;
	}
	
	void push(Object o){
		if(head == null)
			head = o;
		runner = head;
		while(runner.next!=null){
			runner = runner.next;
		}
		runner.next = o;
		
	}
}
